def fecharTodasAsJanelas():
    sistema_operacional = platform.system()

    if sistema_operacional == "Windows":
        # Encerra o processo do Chrome no Windows
        os.system("taskkill /f /im chrome.exe")
    elif sistema_operacional == "Linux" or sistema_operacional == "Darwin":  # Darwin é o macOS
        # Encerra o processo do Chrome no Linux/macOS
        os.system("pkill chrome")
    else:
        print("Sistema operacional não suportado.")

def encontrar_csv_mais_recente(caminho_pasta):
    try:
        # Listar todos os arquivos no diretório com extensão .csv
        arquivos_csv = [f for f in os.listdir(caminho_pasta) if f.endswith('.csv')]

        # Verificar se há pelo menos um arquivo .csv
        if not arquivos_csv:
            return None  # Retorna None se não houver arquivos .csv no diretório

        # Construir caminhos completos para os arquivos .csv
        caminhos_csv = [os.path.join(caminho_pasta, arquivo) for arquivo in arquivos_csv]

        # Encontrar o arquivo .csv mais recente com base na data de modificação
        arquivo_mais_recente = max(caminhos_csv, key=os.path.getmtime)

        arquivo_mais_recente = arquivo_mais_recente.replace("\\", "/")
        #print(type(arquivocsvMaisRecente))
       
        return arquivo_mais_recente

    except Exception as e:
        print(f"Ocorreu um erro: {e}")
        return None

def int_to_roman(num):
    val = [
        2000,1000, 900, 500, 400,
        100, 90, 50, 40,
        10, 9, 5, 4,
        1
    ]
    syb = [
        "MM","M", "CM", "D", "CD",
        "C", "XC", "L", "XL",
        "X", "IX", "V", "IV",
        "I"
    ]
    roman_num = ''
    i = 0
    while num > 0:
        for _ in range(num // val[i]):
            roman_num += syb[i]
            num -= val[i]
        i += 1
    return roman_num

def eh_romano(s):
    pattern = '^M{0,3}(CM|CD|D?C{0,3})(XC|XL|L?X{0,3})(IX|IV|V?I{0,3})$'
    return re.match(pattern, s) is not None

def filtrarRomanos(lst):
    return list(filter(eh_romano, lst))


def romano_para_inteiro(roman):
    #print('roman ->', roman)
    numeraisRomanos = {'I': 1, 'V': 5, 'X': 10, 'L': 50, 'C': 100, 'D': 500, 'M': 1000}
    resultado = 0
    valorprevio = 0

    for numeral in reversed(roman):
        valor = numeraisRomanos[numeral]
        if valor < valorprevio:
            resultado -= valor
        else:
            resultado += valor
        valorprevio = valor

    return resultado

def roman_to_int(roman):
    roman_numerals = {'I': 1, 'V': 5, 'X': 10, 'L': 50, 'C': 100, 'D': 500, 'M': 1000}
    result = 0
    prev_value = 0

    for numeral in reversed(roman):
        value = roman_numerals[numeral]
        if value < prev_value:
            result -= value
        else:
            result += value
        prev_value = value

    return result

def eh_romano(s):
    roman_numerals = {'I': 1, 'V': 5, 'X': 10, 'L': 50, 'C': 100, 'D': 500, 'M': 1000}
   
    def is_valid_roman(s):
        invalid_substrings = ['IIII', 'VV', 'XXXX', 'LL', 'CCCC', 'DD', 'MMMM']
        return not any(substring in s for substring in invalid_substrings)

    if not isinstance(s, str) or not s:
        return False

    s = s.upper()

    # Verificar se a string contém apenas caracteres romanos válidos
    if not all(char in roman_numerals for char in s):
        return False

    # Verificar se a string não contém padrões inválidos repetidos
    if not is_valid_roman(s):
        return False

    # Verificar se a string é um número romano válido
    try:
        total = 0
        prev_value = 0

        for numeral in reversed(s):
            value = roman_numerals[numeral]
            if value < prev_value:
                total -= value
            else:
                total += value
            prev_value = value

        return total > 0
    except KeyError:
        return False

def encontrar_id_pelo_nome(html, nome):
    soup = BeautifulSoup(html, 'html.parser')
   
    # Encontrar o elemento <td> com o texto desejado
    elemento_td = soup.find('td', text=lambda x: x and nome in x)
   
    # Verificar se o elemento <td> foi encontrado
    if elemento_td:
        # Encontrar o valor do atributo 'id' no elemento <input> dentro do elemento <td>
        id_do_input = elemento_td.find('input', {'id': True})['id']
        return id_do_input
    else:
        return None



def localizar_chkInfraItem_peloNumeroSEI(html,numeroSEI):
   
    # Parse the HTML content with BeautifulSoup
    soup = BeautifulSoup(html, 'lxml')

    # Find the table with id "tblDocumentos"
    table = soup.find('table', {'id': 'tblDocumentos'})

    # Extract the column headers
    headers = [th.text.strip() for th in table.find_all('th', {'class': 'infraTh'})]

    # Initialize a list to store the data
    data = []

    # Iterate through rows with class "infraTrClara"
    for row in table.find_all('tr', {'class': 'infraTrClara'}):
        # Extract the content of each "td"
        row_data = [td.text.strip() for td in row.find_all('td')]
       
        # Extract the IDs starting with "chkInfraItem" and append to the row_data
        ids = [input_tag['id'] for input_tag in row.find_all('input', {'id': lambda x: x and x.startswith('chkInfraItem')})]
        row_data.append(ids)

        # Append the row data to the main data list
        data.append(row_data)

    # Criar a DataFrame
    df = pd.DataFrame(data, columns=headers + ['chkInfraItemIDs'])

    # Mostrar the DataFrame
    print(df)

def listar_pastas(caminho):
    try:
        # Lista todos os arquivos e pastas no caminho especificado
        conteudo = os.listdir(caminho)

        # Filtra apenas as pastas
        pastas = [item for item in conteudo if os.path.isdir(os.path.join(caminho, item))]

        return pastas
    except Exception as e:
        print(f"Erro ao listar pastas: {e}")
        return []

def remover_pastas_vazias(diretorio):
    if not os.path.isdir(diretorio):
        print(f"O diretório '{diretorio}' não existe.")
        return

    for root, dirs, files in os.walk(diretorio, topdown=False):
        for dir in dirs:
            caminho_pasta = os.path.join(root, dir)
            try:
                if not os.listdir(caminho_pasta):  # Verifica se a pasta está vazia
                    #os.rmdir(caminho_pasta)       # Remove a pasta
                    excluir_pasta_como_administrador(caminho_pasta.replace("/", "\\"))
                    print(f"Pasta vazia removida: {caminho_pasta}")
            except PermissionError:
                print(f"Permissão negada: {caminho_pasta}")
            except Exception as e:
                print(f"Erro ao tentar remover {caminho_pasta}: {e}")

def extrair_texto_pdf_ANTIGO(caminho_arquivo):
    textoTotal = ''
    with open(caminho_arquivo, 'rb') as arquivo_pdf:
        leitor_pdf = PdfReader(arquivo_pdf)
       
        # Lê cada página do PDF e imprime o texto
       
        for pagina_numero in range(len(leitor_pdf.pages)):
            pagina = leitor_pdf.pages[pagina_numero]
            texto = pagina.extract_text()
            textoTotal=textoTotal+texto
            #print(f"Texto da página {pagina_numero + 1}:\n{texto}\n")
    return textoTotal

def extrair_texto_pdf(caminho_arquivo):
    textoTotal = ''
    with open(caminho_arquivo, 'rb') as arquivo_pdf:
        leitor_pdf = PdfReader(arquivo_pdf)
       
        # Lê cada página do PDF e imprime o texto
        if len(leitor_pdf.pages)>3:
            for pagina_numero in range(3):
                pagina = leitor_pdf.pages[pagina_numero]
                texto = pagina.extract_text()
                textoTotal=textoTotal+texto
                #print(f"Texto da página {pagina_numero + 1}:\n{texto}\n")
        else:
            for pagina_numero in range(len(leitor_pdf.pages)):
                pagina = leitor_pdf.pages[pagina_numero]
                texto = pagina.extract_text()
                textoTotal=textoTotal+texto

    return textoTotal





def substituir_pontosDeCondiconantesPorAsterisco(texto):
    # Defina um padrão de expressão regular para encontrar pontos entre números
    padrao = re.compile(r'(\d)\.(\d)')

    # Substitua os pontos pelo caractere '*'
    texto_modificado = padrao.sub(r'\1*\2', texto)

    return texto_modificado

def pegarXapos(text, keyword, max_length):
    # Encontrar a posição da palavra-chave no texto (em minúsculas ou maiúsculas)
    keyword_position = text.lower().find(keyword.lower())

    # Pegar o texto após a palavra-chave e limitá-lo a 300 caracteres
    if keyword_position != -1:
        text = text[keyword_position + len(keyword):]
        remaining_length = len(text)
        if remaining_length > max_length:
            text = text[:max_length]  # Pegar até 300 caracteres, se houver mais
    else:
        text = ""  # Se a palavra-chave não for encontrada, retorne uma string vazia

    return text
